\section{C++ Static Score Function (Logic-First)}
This scoring function is a representative implementation of "correctness-process balance". It gives logic points for key programming constructs and assignment-specific structure, then applies capped penalties for static issues, so syntax mistakes do not fully erase algorithmic intent.

\begin{verbatim}
def _score_cpp_static_analysis(code: str, analysis_summary: dict, assignment_id: int) -> int:
    normalized = (code or "").lower()
    non_empty_lines = [line for line in (code or "").splitlines() if line.strip()]

    logic_points = 0
    if "#include" in normalized:
        logic_points += 8
    if "int main" in normalized:
        logic_points += 12
    if "while" in normalized or "for" in normalized:
        logic_points += 8
    if "return" in normalized:
        logic_points += 4

    # Assignment-specific signals
    if assignment_id == 1:
        if "my_strlen" in normalized:
            logic_points += 16
        if "my_strcpy" in normalized:
            logic_points += 16
    ...

    logic_points = min(logic_points, 70)
    base_score = 25 + logic_points

    error_count = int(analysis_summary.get("error_count", 0) or 0)
    warning_count = int(analysis_summary.get("warning_count", 0) or 0)
    convention_count = int(analysis_summary.get("convention_count", 0) or 0)

    penalty = min(18, error_count*1.5 + warning_count*0.75 + convention_count*0.3)
    score = round(base_score - penalty)

    if logic_points >= 40 and score < 60: score = 60
    if logic_points >= 55 and score < 70: score = 70

    if not non_empty_lines:
        return 0

    return max(0, min(100, score))
\end{verbatim}

                    "expected": expected,

                    "ok": False,
                    "exit_code": 124,
                }
            )

    return {
        "pass_count": sum(1 for c in results if c["ok"]),
        "total": len(results),
        "cases": results,
        "error": None,
    }

\section{Static Analysis Entry (Python/C++ Unified Interface)}
The static analysis module uses a language-dependent path (Pylint for Python, g++ syntax check for C++) and returns a uniform issue summary for downstream scoring and feedback.

\begin{verbatim}
class StaticAnalyzer:
    def analyze(self, code: str, language: str = "python") -> Dict[str, Any]:
        if language == "cpp":
            proc = subprocess.run(
                ["g++", "-std=c++17", "-Wall", "-Wextra", "-fsyntax-only", tmp_path],
                capture_output=True,
                text=True,
                timeout=10,
            )
            stderr = (proc.stderr or "").strip()
            if stderr:
                for raw_line in stderr.splitlines():
                    line = raw_line.strip()
                    if not line:
                        continue
                    issue_type = "warning" if "warning:" in line else "error"
                    issues.append(
                        {
                            "type": issue_type,
                            "message": line,
                            "symbol": "gxx-syntax-check",
                        }
                    )
        else:
            proc = subprocess.run(
                [
                    sys.executable,
                    "-m",
                    "pylint",
                    tmp_path,
                    "--output-format=json",
                    "--score=n",
                ],
                capture_output=True,
                text=True,
                timeout=10,
            )
            output_str = (proc.stdout or "").strip()
            if output_str:
                parsed = json.loads(output_str)
                if isinstance(parsed, list):
                    issues = parsed

        return {
            "issues": issues,
            "error_count": count_type(issues, "error"),
            "warning_count": count_type(issues, "warning"),
            "convention_count": count_type(issues, "convention"),
            "refactor_count": count_type(issues, "refactor"),
        }
\end{verbatim}
